function agm(a::Real, b::Real; tol::Real = 1e-10, maxiter::Int = 100)

    maxiter > 0 || throw(ArgumentError("maximum number of iterations must be positive"))

    end
    return (y1 + y2)/2, abs(y1 - y2), n
end
